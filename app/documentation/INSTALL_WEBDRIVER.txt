
# Install ChromeDriver.
wget -N https://chromedriver.storage.googleapis.com/2.35/chromedriver_linux64.zip -P ~/
unzip ~/chromedriver_linux64.zip -d ~/
rm ~/chromedriver_linux64.zip
sudo mv -f ~/chromedriver /usr/local/bin/chromedriver
sudo chown root:root /usr/local/bin/chromedriver
sudo chmod 0755 /usr/local/bin/chromedriver

# INSTALL MozillaDriver

wget -N https://github.com/mozilla/geckodriver/releases/download/v0.23.0/geckodriver-v0.23.0-linux64.tar.gz -P ~/
tar -xvzf  ~/geckodriver-v0.23.0-linux64.tar.gz 
rm ~/geckodriver-v0.23.0-linux64.tar.gz 
sudo mv -f geckodriver /usr/local/bin/
sudo chown root:root /usr/local/bin/geckodriver 
sudo chmod 0755 /usr/local/bin/geckodriver


# Install Selenium.
wget -N http://selenium-release.storage.googleapis.com/3.8/selenium-server-standalone-3.8.1.jar -P ~/
sudo mv -f ~/selenium-server-standalone-3.8.1.jar /usr/local/bin/selenium-server-standalone.jar
sudo chown root:root /usr/local/bin/selenium-server-standalone.jar
sudo chmod 0755 /usr/local/bin/selenium-server-standalone.jar
